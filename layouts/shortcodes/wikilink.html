{{- $raw := .Get 0 | default "" -}} {{- $parts := split $raw "|" -}} {{- $targetSlug := lower (index $parts 0 | default "") -}} {{- $explicitLabel := gt (len $parts) 1 -}} {{- $linkText := cond $explicitLabel (index $parts 1 | default "") (index $parts 0 | default "") -}} {{- $stemIndex := site.Store.Get "wikilink-stem-index" -}} {{- $titleIndex := site.Store.Get "wikilink-title-index" -}} {{- if not $stemIndex -}} {{- $stemIndex = dict -}} {{- $titleIndex = dict -}} {{- range site.RegularPages -}} {{- $stem := .File.ContentBaseName | lower -}} {{- if ne $stem "" -}} {{- if index $stemIndex $stem -}} {{- warnf "wikilink: duplicate stem %q — %s and %s" $stem (index $stemIndex $stem).File.Path .File.Path -}} {{- end -}} {{- $stemIndex = merge $stemIndex (dict $stem .) -}} {{- end -}} {{- $title := .Title | lower -}} {{- if ne $title "" -}} {{- $titleIndex = merge $titleIndex (dict $title .) -}} {{- end -}} {{- end -}} {{- site.Store.Set "wikilink-stem-index" $stemIndex -}} {{- site.Store.Set "wikilink-title-index" $titleIndex -}} {{- end -}} {{- $found := index $stemIndex $targetSlug | default (index $titleIndex $targetSlug) | default "" -}} {{- if ne $found "" -}} {{- if not $explicitLabel -}}{{- $linkText = $found.Title -}}{{- end -}} {{ $linkText }} {{- else -}} {{ $raw }} {{- end -}}
